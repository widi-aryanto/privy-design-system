  import { showOverlay, hideOverlay } from '.'
  import Button from '../button/Button.vue'
  import Overlay from './Overlay.vue'

  const overlay = ref()

  function onClick() {
    showOverlay()

    setTimeout(() => {
      hideOverlay()
    }, 3000)
  }

  onMounted(() => {
    overlay.value.show()
  })
